# Connect to Hive 
hive
# Create database 
create database hive_class_b1;
# Connect to the database 
use hive_class_b1;
# Create table in database
create table department_data                                                                                                            
    > (                                                                                                                                       
    > dept_id int,                                                                                                                            
    > dept_name string,                                                                                                                       
    > manager_id int,                                                                                                                         
    > salary int)                                                                                                                             
    > row format delimited                                                                                                                    
    > fields terminated by ','; 
# Show tables
show tables
# Describe table
describe department_data 
describe formatted department_table ;
